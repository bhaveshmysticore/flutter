Flutter MVVM Architecture Example

1. What is MVVM?

MVVM means:

  • Model → Data structure
  • View → UI Screen
  • ViewModel → Business Logic

This architecture keeps the code clean and maintainable.

3. Model (Product Model)

This represents API data.

class Product {

final int id;
final String title;
final String image;

Product({
required this.id,
required this.title,
required this.image
});

factory Product.fromJson(Map json){

return Product(
id: json["id"],
title: json["title"],
image: json["image"]
);

}

}

4. API Service

This file is responsible for calling API.

import 'dart:convert';
import 'package:http/http.dart' as http;
import '../models/product_model.dart';

class ApiService {

Future> fetchProducts() async {

final response = await http.get(
Uri.parse("https://fakestoreapi.com/products")
);

if(response.statusCode == 200){

List data = jsonDecode(response.body);

return data.map((e)=>Product.fromJson(e)).toList();

}else{

throw Exception("Failed to load products");

}

}

}

5. Local Database (SQLite)

This stores products locally.

import 'package:sqflite/sqflite.dart';
import 'package:path/path.dart';

class DatabaseService {

static Database? _db;

Future get database async {

if(_db != null) return _db!;

_db = await initDB();

return _db!;

}

Future initDB() async {

final path = join(await getDatabasesPath(),"product.db");

return openDatabase(
path,
version:1,
onCreate:(db,version){

db.execute(
"CREATE TABLE products(id INTEGER PRIMARY KEY,title TEXT,image TEXT)"
);

}
);

}

}

6. ViewModel

This connects API and UI.

import 'package:flutter/material.dart';
import '../models/product_model.dart';
import '../services/api_service.dart';

class ProductViewModel extends ChangeNotifier{

List products = [];

bool loading = false;

final ApiService api = ApiService();

Future loadProducts() async{

loading = true;
notifyListeners();

products = await api.fetchProducts();

loading = false;
notifyListeners();

}

}

7. List Screen

class ProductListScreen extends StatelessWidget {

@override
Widget build(BuildContext context) {

return Scaffold(

appBar: AppBar(title: Text("Product List")),

body: Consumer(
builder:(context,vm,child){

if(vm.loading){

return Center(child:CircularProgressIndicator());

}

return ListView.builder(

itemCount: vm.products.length,

itemBuilder:(context,index){

var product = vm.products[index];

return ListTile(

leading: Image.network(product.image,width:40),

title: Text(product.title),

);

});

}

)

);

}

}

8. Grid Screen

class ProductGridScreen extends StatelessWidget {

@override
Widget build(BuildContext context) {

return Scaffold(

appBar: AppBar(title: Text("Product Grid")),

body: Consumer(

builder:(context,vm,child){

return GridView.builder(

gridDelegate:
SliverGridDelegateWithFixedCrossAxisCount(
crossAxisCount:2
),

itemCount: vm.products.length,

itemBuilder:(context,index){

var product = vm.products[index];

return Card(

child: Column(

children:[

Image.network(product.image,height:80),

Text(product.title)

]

)

);

}

);

}

)

);

}

}

9. main.dart

void main(){

}

class MyApp extends StatelessWidget {

@override
Widget build(BuildContext context) {

}

11. Data Flow in MVVM

The application flow works like this:

View (UI Screen)
     ↓
ViewModel (Logic)
     ↓
API Service / Database
     ↓
Model (Data)

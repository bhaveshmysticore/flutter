Flutter MVVM Clean Architecture Complete Guide

This document teaches how large production Flutter apps are built using MVVM + Clean Architecture.

1. What is Clean Architecture

Clean Architecture separates code into independent layers. Each layer has a specific responsibility.

  • Presentation Layer
  • Domain Layer
  • Data Layer

Benefits:

  • Easy testing
  • Maintainable code
  • Scalable projects
  • Team collaboration

2. MVVM Architecture

MVVM stands for:

  • Model → Data layer
  • View → UI layer
  • ViewModel → Business logic
Architecture flow: 
UI (View)
   ↓
ViewModel
   ↓
Repository
   ↓
Data Source (API / Database)

6. API Client


import 'package:dio/dio.dart';

class ApiClient {

final Dio dio = Dio();

Future get(String url) async {

try{

return await dio.get(url);

}catch(e){

throw Exception("API Error");

}

}

}

9. Repository Interface


abstract class ProductRepository {

Future> getProducts();

}

10. Repository Implementation


class ProductRepositoryImpl implements ProductRepository {

final ApiClient apiClient;

ProductRepositoryImpl(this.apiClient);

@override
Future> getProducts() async {

final response = await apiClient.get("/products");

List list = response.data;

return list.map((e)=>Product(e["id"],e["name"])).toList();

}

}

11. Use Case


class GetProducts {

final ProductRepository repository;

GetProducts(this.repository);

Future> call(){

return repository.getProducts();

}

}

12. ViewModel


import 'package:flutter/material.dart';

class ProductViewModel extends ChangeNotifier {

final GetProducts getProducts;

ProductViewModel(this.getProducts);

List products = [];

bool loading = false;

Future loadProducts() async{

loading = true;
notifyListeners();

products = await getProducts();

loading = false;
notifyListeners();

}

}

13. UI Screen


class ProductScreen extends StatelessWidget {

@override
Widget build(BuildContext context){

return Scaffold(

appBar: AppBar(title: Text("Products")),

body: Consumer(

builder:(context,vm,_){

if(vm.loading) return CircularProgressIndicator();

return ListView.builder(

itemCount: vm.products.length,

itemBuilder:(context,index){

final p = vm.products[index];

return ListTile(title: Text(p.name));

});

}

)

);

}

}
